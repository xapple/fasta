 28class FASTQ(FASTA):
 29    """
 30    A single FASTQ file somewhere in the filesystem.
 31    It is very similar to a FASTA file object, except we have a few added
 32    methods and properties for dealing with the quality scores.
 33    """
 34
 35    ext    = 'fastq'
 36    format = 'fastq'
 37
 38    #----------------------------- Properties --------------------------------#
 39    @property_cached
 40    def count(self):
 41        # Import module #
 42        from shell_command import shell_output
 43        # Case when we are not compressed #
 44        if not self.gzipped:
 45            return int(int(shell_output("cat %s | wc -l" % self.path)) / 4)
 46        # If we are gzipped we can just use zcat or gzcat on macOS #
 47        program = 'gzcat' if sys.platform != 'linux' else 'zcat'
 48        command = "%s %s | wc -l" % (program, self.path)
 49        return int(int(shell_output(command)) / 4)
 50
 51    @property_cached
 52    def avg_quality(self):
 53        """
 54        Returns a single float.
 55        Quite slow and inefficient computationally.
 56        Think of a better implementation?
 57        """
 58        mean = average(s for r in self for s in
 59                       r.letter_annotations["phred_quality"])
 60        return mean
 61
 62    #----------------------------- Conversion --------------------------------#
 63    def to_fasta(self, path, verbose=False):
 64        # Select verbosity #
 65        import tqdm
 66        wrapper = tqdm.tqdm if verbose else lambda x: x
 67        # Do it #
 68        with open(path, 'w') as handle:
 69            for r in wrapper(self): SeqIO.write(r, handle, 'fasta')
 70        # Return #
 71        return FASTA(path)
 72
 73    def to_qual(self, path, verbose=False):
 74        # Select verbosity #
 75        import tqdm
 76        wrapper = tqdm.tqdm if verbose else lambda x: x
 77        # Do it #
 78        with open(path, 'w') as handle:
 79            for r in wrapper(self): SeqIO.write(r, handle, 'qual')
 80        # Return #
 81        return FilePath(path)
 82
 83    #-------------------------------- Tools ----------------------------------#
 84    @property_cached
 85    def validator(self):
 86        """Validate the format of this FASTQ."""
 87        from fasta.validator import Validator
 88        return Validator(self.path)
 89
 90    @property_cached
 91    def fastqc(self):
 92        """Run the FastQC software on this FASTQ."""
 93        from fasta.fastqc import FastQC
 94        return FastQC(self)
 95
 96    #-------------------------------- PHRED Format ---------------------------#
 97    def guess_phred_format(self):
 98        """
 99        Guess the PHRED score format. The gold standard is the first one, aka
100        the one called "Sanger". Sanger encoding is exactly equivalent to
101        Illumina-1.8 encoding.
102        In other words, they finally gave up with their alternative standards.
103        """
104        # Possibilities #
105        self.intervals = {
106            'Sanger':       (33,  74),    # <- This one is the gold standard
107            'Solexa':       (59, 105),    # <- This one is rare
108            'Illumina-1.3': (64, 105),    # <- These were abandoned after they wised up.
109            'Illumina-1.5': (67, 105),    # <- These were abandoned after they wised up.
110        }
111        # Initialize variables #
112        glob_min  = 9999
113        glob_max  = 0
114        take_next = False
115        # Stop after #
116        max_sequences_to_consider = 2000000
117        # Error message #
118        msg = "Multiple PHRED encodings possible for '%s'.\n"
119        # Loop #
120        self.open()
121        for i, line in enumerate(self.handle):
122            # Are we on the PHRED line? #
123            if not take_next:
124                take_next = line.startswith('+')
125                continue
126            # Now we are on the PHRED line! #
127            take_next = False
128            current_min, current_max = self.get_qual_range(line.rstrip('\n'))
129            if current_min < glob_min or current_max > glob_max:
130                glob_min = min(current_min, glob_min)
131                glob_max = max(current_max, glob_max)
132                valid_encodings = [e for e, r in self.intervals.items()
133                                   if glob_min >= r[0] and glob_max <= r[1]]
134                if len(valid_encodings) == 0:
135                    msg = "Illegal PHRED encoding for '%s'.\n"
136                    break
137                if len(valid_encodings) == 1:
138                    return valid_encodings[0]
139            # Stop condition #
140            if i >= max_sequences_to_consider: break
141        # It didn't work #
142        msg += "Maximum detected value is: %s.\n"
143        msg += "Minimum detected value is: %s.\n"
144        msg += "Possibilities are: %s."
145        msg  = msg % (self, glob_max, glob_min, ' or '.join(valid_encodings))
146        raise Exception(msg)
147
148    def get_qual_range(self, phred_string):
149        """
150        >>> self.get_qual_range("DLXYXXRXWYYTPMLUUQWTXTRSXSWMDMTRNDNSMJFJFFRMV")
151        (68, 89)
152        """
153        values = [ord(char) for char in phred_string]
154        return min(values), max(values)
155
156    #------------------------- Conversion methods ----------------------------#
157    def phred_13_to_18(self, new_path=None, in_place=True):
158        """Illumina-1.3 format conversion to Illumina-1.8 format via BioPython."""
159        # New file #
160        if new_path is None:
161            new_fastq = self.__class__(new_temp_path(suffix=self.extension))
162        else:
163            new_fastq = self.__class__(new_path)
164        # Do it #
165        self.format = 'fastq-illumina'
166        new_fastq.open('w')
167        new_fastq.handle.writelines(seq.format('fastq-sanger') for seq in self)
168        new_fastq.close()
169        self.format = 'fastq-sanger'
170        # Return #
171        if in_place:
172            os.remove(self.path)
173            shutil.move(new_fastq, self.path)
174            return self
175        else: return new_fastq
176
177    def phred_13_to_18_sed(self, new_path=None, in_place=True):
178        """Illumina-1.3 format conversion to Illumina-1.8 format via sed (faster)."""
179        # String #
180        sed_command = r"""4~4y/@ABCDEFGHIJKLMNOPQRSTUVWXYZ[\\]^_`abcdefghi/!"#$%&'\''()*+,-.\/0123456789:;<=>?@ABCDEFGHIJ/"""
181        # Faster with bash utilities #
182        if in_place is True:
183            sh.sed('-i', sed_command, self.path)
184            return self
185        # New file #
186        if new_path is None: new_fastq = self.__class__(new_temp_path())
187        else:                new_fastq = self.__class__(new_path)
188        sh.sed(sed_command + " " + new_fastq, self.path)
189        return new_fastq
